<script>

document.addEventListener('DOMContentLoaded', function () {

    // Mobile Menu

    const mobileMenuButton = document.getElementById('mobile-menu-button');

    const mobileMenu = document.getElementById('mobile-menu');

    mobileMenuButton.addEventListener('click', () => {

        mobileMenu.classList.toggle('hidden');

        mobileMenu.classList.toggle('flex');

    });


    document.querySelectorAll('#mobile-menu a').forEach(link => {

        link.addEventListener('click', () => {

            mobileMenu.classList.add('hidden');

            mobileMenu.classList.remove('flex');

        });

    });


    // Auto-save functions

    function saveState() {

        const state = {

            reflectionQuestion: document.getElementById('reflectionQuestion').value,

            biggestFear: document.getElementById('biggestFear').value,

            dreamLifeVision_1: document.getElementById('dreamLifeVision_1').value,

            dreamLifeVision_2: document.getElementById('dreamLifeVision_2').value,

            dreamLifeVision_3: document.getElementById('dreamLifeVision_3').value,

            postedWeeks: window.postedWeeks, // Access the global variable

            practiceLog: [],

            mindsetShift: [],

        };


        // Save practice log checkboxes

        document.querySelectorAll('#practice-log input[type="checkbox"]').forEach((checkbox, index) => {

            state.practiceLog.push(checkbox.checked);

        });


        // Save mindset shift checkboxes

        document.querySelectorAll('#mindset-checklist input[type="checkbox"]').forEach((checkbox, index) => {

            state.mindsetShift.push(checkbox.checked);

        });


        localStorage.setItem('tiktokMindsetSPAState', JSON.stringify(state));

    }


    function loadState() {

        const savedState = localStorage.getItem('tiktokMindsetSPAState');

        if (savedState) {

            const state = JSON.parse(savedState);


            document.getElementById('reflectionQuestion').value = state.reflectionQuestion || '';

            document.getElementById('biggestFear').value = state.biggestFear || '';

            document.getElementById('dreamLifeVision_1').value = state.dreamLifeVision_1 || '';

            document.getElementById('dreamLifeVision_2').value = state.dreamLifeVision_2 || '';

            document.getElementById('dreamLifeVision_3').value = state.dreamLifeVision_3 || '';


            // Load practice log checkboxes

            document.querySelectorAll('#practice-log input[type="checkbox"]').forEach((checkbox, index) => {

                if (state.practiceLog[index] !== undefined) {

                    checkbox.checked = state.practiceLog[index];

                    checkbox.dispatchEvent(new Event('change')); // Trigger change to update styling

                }

            });


            // Load mindset shift checkboxes

            document.querySelectorAll('#mindset-checklist input[type="checkbox"]').forEach((checkbox, index) => {

                if (state.mindsetShift[index] !== undefined) {

                    checkbox.checked = state.mindsetShift[index];

                    checkbox.dispatchEvent(new Event('change')); // Trigger change to update styling

                }

            });


            window.postedWeeks = state.postedWeeks || 0; // Set global variable

        }

    }


    // Initialize/Load state on page load

    loadState();



    // Section 1: Practice Log

    const practiceLogContainer = document.getElementById('practice-log');

    for (let i = 1; i <= 7; i++) {

        const div = document.createElement('div');

        div.className = 'flex items-center bg-gray-100 p-3 rounded-lg';

        div.innerHTML = `

            <input type="checkbox" id="day-${i}" class="h-5 w-5 rounded mr-3 accent-teal">

            <label for="day-${i}" class="flex-grow interactive-checklist-item">Day ${i}: Practice Session</label>

        `;

        practiceLogContainer.appendChild(div);

    }

    

    // Checklist item styling and save

    document.querySelectorAll('.interactive-checklist-item').forEach(item => {

        const checkbox = item.previousElementSibling;

        checkbox.addEventListener('change', () => {

            item.classList.toggle('completed', checkbox.checked);

            saveState(); // Save state on checkbox change

        });

    });


    const mindsetChecklist = document.getElementById('mindset-checklist');

    mindsetChecklist.querySelectorAll('input[type="checkbox"]').forEach(checkbox => {

        checkbox.addEventListener('change', () => {

            checkbox.nextElementSibling.classList.toggle('completed', checkbox.checked);

            saveState(); // Save state on checkbox change

        });

    });


    // Text input auto-save

    document.getElementById('reflectionQuestion').addEventListener('input', saveState);

    document.getElementById('biggestFear').addEventListener('input', saveState);

    document.getElementById('dreamLifeVision_1').addEventListener('input', saveState);

    document.getElementById('dreamLifeVision_2').addEventListener('input', saveState);

    document.getElementById('dreamLifeVision_3').addEventListener('input', saveState);



    // Section 3: Consistency Chart

    const ctx = document.getElementById('consistencyChart').getContext('2d');

    const totalWeeks = 12;

    window.postedWeeks = window.postedWeeks !== undefined ? window.postedWeeks : 0; // Initialize or load

    

    const chartLabels = Array.from({ length: totalWeeks }, (_, i) => `W${i + 1}`);

    const chartData = {

        labels: chartLabels,

        datasets: [{

            label: 'Weeks Posted',

            data: Array(totalWeeks).fill(0),

            backgroundColor: (context) => {

                const index = context.dataIndex;

                return index < window.postedWeeks ? '#4ECDC4' : '#E5E7EB';

            },

            borderColor: '#FFF',

            borderWidth: 2,

            borderRadius: 5,

        }]

    };


    const consistencyChart = new Chart(ctx, {

        type: 'bar',

        data: chartData,

        options: {

            responsive: true,

            maintainAspectRatio: false,

            plugins: {

                legend: {

                    display: false,

                },

                tooltip: {

                    enabled: false

                }

            },

            scales: {

                y: {

                    display: false,

                    beginAtZero: true,

                    max: 1,

                },

                x: {

                    grid: {

                        display: false,

                    },

                     ticks: {

                        font: {

                            size: 10

                        }

                    }

                }

            }

        }

    });


    const addWeekButton = document.getElementById('addWeekButton');

    const resetChartButton = document.getElementById('resetChartButton');


    function updateChart() {

        consistencyChart.data.datasets[0].data = Array(totalWeeks).fill(0).map((_, i) => i < window.postedWeeks ? 1 : 0);

        consistencyChart.update();

        saveState(); // Save state after chart update

    }

    

    addWeekButton.addEventListener('click', () => {

        if (window.postedWeeks < totalWeeks) {

            window.postedWeeks++;

            updateChart();

        }

    });


    resetChartButton.addEventListener('click', () => {

        window.postedWeeks = 0;

        updateChart();

    });

    

    updateChart(); // Initial chart update after loading state

});

</script>
